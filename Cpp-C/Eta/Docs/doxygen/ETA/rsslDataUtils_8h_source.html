00042 RTR_C_ALWAYS_INLINE RsslInt32 rsslCompareString(const RsslBuffer *pBuf1, const RsslBuffer *pBuf2)
00043 {
00044     RsslUInt32 len;
00045 
00046     /* Get the shorter length */
00047     len = (pBuf1->length < pBuf2->length) ? pBuf1->length : pBuf2->length;
00048 
00049     /* Match the ends of the strings -- either they should be the same length,
00050      * or the longer one should have a null-terminator where the other one ends. */
00051     if (pBuf1->length > len && pBuf1->data[len] != '\0') return 1;
00052     if (pBuf2->length > len && pBuf2->data[len] != '\0') return -1;
00053     
00054     /* If the above succeeded, strnicmp is sufficient because it will compare 
00055      * all relevant characters (or up to null-terminators if they are found). */
00056     return strncmp(pBuf1->data, pBuf2->data, len);
00057 }
